-- [[ UTILITIES ]] --
local function GetBouncingColor()
    return Color3.fromRGB(0, 0, 0):Lerp(Color3.fromRGB(255, 0, 0), (math.sin(tick() * 2) + 1) / 2)
end

local function SmartPressE(prompt)
    if not prompt or not prompt.Enabled then return end
    task.spawn(function()
        prompt:InputHoldBegin()
        task.wait((prompt.HoldDuration or 0) + 0.1)
        prompt:InputHoldEnd()
        fireproximityprompt(prompt)
    end)
end

local function FlyTo(targetCFrame, speed)
    local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    local distance = (hrp.Position - targetCFrame.Position).Magnitude
    local info = TweenInfo.new(distance / speed, Enum.EasingStyle.Linear)
    local tween = TweenService:Create(hrp, info, {CFrame = targetCFrame})
    tween:Play()
    tween.Completed:Wait()
end

-- [[ BYTENET SYSTEM (GLOCK19) ]] --
local function SaveGlock()
    pcall(function()
        local args = { buffer.fromstring("\003\006\000Change\a\000Glock19\b\000Backpack") }
        ReplicatedStorage:WaitForChild("Package"):WaitForChild("ByteNetMax"):WaitForChild("system"):WaitForChild("ByteNetReliable"):FireServer(unpack(args))
    end)
end

local function LoadGlock()
    pcall(function()
        task.wait(1.2)
        local args = { buffer.fromstring("\003\006\000Change\a\000Glock19\003\000Inv") }
        ReplicatedStorage:WaitForChild("Package"):WaitForChild("ByteNetMax"):WaitForChild("system"):WaitForChild("ByteNetReliable"):FireServer(unpack(args))
    end)
end

-- ระบบ Loop ตรวจจับการตาย
local function MonitorLifeCycle(character)
    local humanoid = character:WaitForChild("Humanoid")
    humanoid.Died:Connect(function()
        if states.autoSaveGlock then
            SaveGlock() 
            LocalPlayer.CharacterAdded:Wait() 
            LoadGlock() 
        end
    end)
end

if LocalPlayer.Character then task.spawn(MonitorLifeCycle, LocalPlayer.Character) end
LocalPlayer.CharacterAdded:Connect(function(char) task.spawn(MonitorLifeCycle, char) end)

-- [[ AUTO FARM SYSTEM ]] --
task.spawn(function()
    while true do
        if states.autoFarm then
            pcall(function()
                local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    FlyTo(CFrame.new(FARM_CENTER), 100)
                    local farmFolder = workspace:FindFirstChild("Map") and workspace.Map:FindFirstChild("จุดฟาร์ม")
                    if farmFolder then
                        local targets = {}
                        local validNames = {["MachineBase"]=true, ["Microwave"]=true, ["Fridge"]=true}
                        for _, v in pairs(farmFolder:GetChildren()) do
                            if validNames[v.Name] then
                                local dynamicCooldown = math.random(20, 25)
                                if not farmCooldowns[v] or (tick() - farmCooldowns[v]) > dynamicCooldown then
                                    local pos = v:GetPivot().Position
                                    if (pos - Vector3.new(562, 13, -865)).Magnitude < 35 or (pos - Vector3.new(549, 13, -831)).Magnitude < 35 then 
                                        table.insert(targets, v) 
                                    end
                                end
                            end
                        end
                        if #targets > 0 then
                            local selected = targets[math.random(1, #targets)]
                            farmCooldowns[selected] = tick()
                            FlyTo(selected:GetPivot() * CFrame.new(0, 2, 4), 50)
                            local prompt = selected:FindFirstChildWhichIsA("ProximityPrompt", true)
                            if prompt then task.wait(0.2); SmartPressE(prompt); task.wait((prompt.HoldDuration or 0) + 0.6) end
                        end
                    end
                end
            end)
        end
        task.wait(0.5)
    end
end)

-- [[ ESP SYSTEM ]] --
local activeTracers, activeLabels = {}, {}
local function RemoveDrawingESP(player)
    if activeTracers[player] then activeTracers[player]:Remove(); activeTracers[player] = nil end
    if activeLabels[player] then activeLabels[player]:Remove(); activeLabels[player] = nil end
end
local function CreateDrawingESP(player)
    RemoveDrawingESP(player)
    local line = Drawing.new("Line"); line.Visible = false; line.Color = Color3.fromRGB(255, 0, 0); line.Thickness = 1; activeTracers[player] = line
    local text = Drawing.new("Text"); text.Visible = false; text.Color = Color3.fromRGB(255, 255, 255); text.Size = 14; text.Center = true; text.Outline = true; activeLabels[player] = text
end

-- [[ SYSTEM LOOPS ]] --
RunService.RenderStepped:Connect(function()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("Head") then
            local head = player.Character.Head
            local vector, onScreen = Camera:WorldToViewportPoint(head.Position)
            if activeTracers[player] then
                activeTracers[player].Visible = states.espLine and onScreen
                if activeTracers[player].Visible then activeTracers[player].From = Vector2.new(Camera.ViewportSize.X/2, 0); activeTracers[player].To = Vector2.new(vector.X, vector.Y) end
            end
            if activeLabels[player] then
                activeLabels[player].Visible = (states.espName or states.espDisplayName) and onScreen
                if activeLabels[player].Visible then activeLabels[player].Position = Vector2.new(vector.X, vector.Y - 25); activeLabels[player].Text = states.espDisplayName and player.DisplayName or player.Name end
            end
        end
    end
end)

RunService.Heartbeat:Connect(function()
    if states.autoPrompt and not states.autoFarm then
        for _, v in pairs(workspace:GetDescendants()) do if v:IsA("ProximityPrompt") and v.Enabled then fireproximityprompt(v) end end
    end
    if states.bouncingHead then
        local color = GetBouncingColor()
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Head") then
                p.Character.Head.Size, p.Character.Head.Color, p.Character.Head.CanCollide = Vector3.new(5,5,5), color, false
            end
        end
    end
end)
