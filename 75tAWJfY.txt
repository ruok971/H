-- [[ UTILITY FUNCTIONS ]] --
local function GetColorFromRarity(rarityName)
    local colors = {
        ['Common'] = Color3.fromRGB(255, 255, 255),
        ['UnCommon'] = Color3.fromRGB(99, 255, 52),
        ['Rare'] = Color3.fromRGB(51, 170, 255),
        ['Legendary'] = Color3.fromRGB(255, 150, 0),
        ['Epic'] = Color3.fromRGB(237, 44, 255),
        ['Omega'] = Color3.fromRGB(255, 20, 51),
    }
    return colors[rarityName] or Color3.fromRGB(255, 255, 255)
end

local function isPlayerExcluded(plr)
    if not plr or plr == player then return true end
    if plr:GetAttribute("SilentAimIgnore") then return true end
    return table.find({"MithnoFs_49"}, plr.Name) ~= nil
end

local function CreateDrawings()
    if SilentFOVCircle then pcall(function() SilentFOVCircle:Remove() end) end
    SilentFOVCircle = Drawing.new("Circle")
    SilentFOVCircle.Thickness, SilentFOVCircle.NumSides, SilentFOVCircle.Transparency = 2, 64, 0.8
    tracerLine = Drawing.new("Line")
    tracerLine.Thickness, tracerLine.Color = 1, Color3.new(1, 0.2, 0.2)
    targetDot = Drawing.new("Circle")
    targetDot.Radius, targetDot.Filled, targetDot.Color = 4, true, Color3.new(1, 0.2, 0.2)
end

local function GetTarget()
    local closest, shortestDistance = nil, states.FOV
    local center = Camera.ViewportSize / 2
    for _, plr in pairs(Players:GetPlayers()) do
        if not isPlayerExcluded(plr) and plr.Character then
            local part = plr.Character:FindFirstChild(states.LockPart)
            local hum = plr.Character:FindFirstChild("Humanoid")
            if part and hum and hum.Health > 0 then
                local screenPos, onScreen = Camera:WorldToViewportPoint(part.Position)
                local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                if onScreen and dist < shortestDistance then
                    shortestDistance, closest = dist, plr
                end
            end
        end
    end
    return closest
end

        local args = {...}
        if states.SilentAimEnabled and args[2] == "shoot_gun" and self == Remote then
            local target = GetTarget()
            if target and target.Character:FindFirstChild(states.LockPart) then
                local aimPart = target.Character[states.LockPart]
                local aimPos = aimPart.Position + (aimPart.Parent.PrimaryPart.Velocity * 0.15)
                args[4] = CFrame.new(1/0, 1/0, 1/0)
                args[5] = {[1] = {[1] = {Instance = aimPart, Position = aimPos}}}
            end
        end
        return oldFire(self, unpack(args))
    end)
end

        end
    end

    if states.antiDeathV2 and humanoid.Health < 16 then char.HumanoidRootPart.CFrame *= CFrame.new(0, -12, 0) end
