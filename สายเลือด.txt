local function showNotify(text, duration)
    local frame = Instance.new("Frame", holder)
    frame.Size = UDim2.new(0, 0, 0, 60)
    frame.BackgroundColor3 = Color3.fromRGB(20, 20, 25)
    frame.ClipsDescendants = true
    
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 8)
    local stroke = Instance.new("UIStroke", frame)
    stroke.Thickness = 1.5
    stroke.Color = Color3.fromRGB(0, 255, 150)

    local label = Instance.new("TextLabel", frame)
    label.Size = UDim2.new(1, -20, 1, -5)
    label.Position = UDim2.new(0, 15, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.fromRGB(240, 240, 240)
    label.TextSize = 14
    label.Font = Enum.Font.GothamBold
    label.TextXAlignment = Enum.TextXAlignment.Left

    local bar = Instance.new("Frame", frame)
    bar.Size = UDim2.new(1, 0, 0, 3)
    bar.Position = UDim2.new(0, 0, 1, -3)
    bar.BackgroundColor3 = Color3.fromRGB(0, 255, 150)
    bar.BorderSizePixel = 0

    TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Back), {Size = UDim2.new(1, 0, 0, 60)}):Play()
    TweenService:Create(bar, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Size = UDim2.new(0, 0, 0, 3)}):Play()

    task.delay(duration, function()
        local t = TweenService:Create(frame, TweenInfo.new(0.4), {Size = UDim2.new(0, 0, 0, 60), BackgroundTransparency = 1})
        t:Play()
        t.Completed:Connect(function() frame:Destroy() end)
    end)

    return {set_text = function(t) label.Text = t end}
end

-- [[ 3. LOGIC FUNCTIONS ]]
local function walkToTarget(targetPos)
    local path = PathfindingService:CreatePath({AgentRadius = 2, AgentHeight = 5})
    local success, _ = pcall(function() path:ComputeAsync(rootPart.Position, targetPos) end)
    if success and path.Status == Enum.PathStatus.Success then
        for _, waypoint in pairs(path:GetWaypoints()) do
            if not isRunning then break end
            local targetCFrame = CFrame.new(waypoint.Position + Vector3.new(0, 2, 0))
            local dist = (targetCFrame.Position - rootPart.Position).Magnitude
            local t = TweenService:Create(rootPart, TweenInfo.new(dist/35, Enum.EasingStyle.Linear), {CFrame = targetCFrame})
            t:Play()
            t.Completed:Wait()
        end
    end
end

local function getNewTarget()
    local success, folder = pcall(function() return workspace.Job["งานก่อสร้าง"]["งาน"]["หิน"] end)
    if not success or not folder then return nil end
    local items = {}
    for _, v in pairs(folder:GetChildren()) do
        if v.Name == "DrillSite" and not doneList[v] then table.insert(items, v) end
    end
    return #items > 0 and items[math.random(1, #items)] or nil
end

local function startDrilling(duration)
    local start = tick()
    while (tick() - start) < duration and isRunning do
        for _, gui in pairs(player.PlayerGui:GetDescendants()) do
            if gui:IsA("ImageButton") and gui.Visible and gui.AbsolutePosition.Y > 100 then
                local pos = gui.AbsolutePosition + (gui.AbsoluteSize / 2)
                VirtualInputManager:SendMouseButtonEvent(pos.X, pos.Y, 0, true, game, 1)
            end
        end
        task.wait(0.5)
    end
    VirtualInputManager:SendMouseButtonEvent(0, 0, 0, false, game, 1)
end

-- [[ 4. MAIN LOOP ]]
task.spawn(function()
    walkToTarget(startPoint)
    
    local notify = showNotify("ระบบพร้อมทำงานใน 15 วินาที", 15)
    for i = 15, 1, -1 do
        pcall(function() notify.set_text("โปรดรับงาน (" .. i .. " วินาที)") end)
        task.wait(1)
    end

    showActionMenu() -- ล็อคจอและแสดงเมนู

    while isRunning do
        local target = getNewTarget()
        if target then
            walkToTarget(target:GetPivot().Position)
            local prompt = target:FindFirstChildWhichIsA("ProximityPrompt", true)
            if prompt then
                fireproximityprompt(prompt)
                task.wait(1)
                startDrilling(30)
                doneList[target] = true
                task.delay(20, function() doneList[target] = nil end)
            end
        else
            task.wait(1)
        end
        task.wait(0.1)
    end
end)
